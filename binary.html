               stateSpan.textContent = 'X';
               stateSpan.classList.add('text-gray-500');
           }

           bitGroupDiv.appendChild(weightSpan);
           bitGroupDiv.appendChild(wireContainer);
           bitGroupDiv.appendChild(stateSpan);

           wireGrid.appendChild(bitGroupDiv);
       }

       // 調整輸入框最大值
       updateMaxDecimalValue(currentWireCount);
   };
   // 更新十進位輸入框的最大值
   const updateMaxDecimalValue = (count) => {
       const maxValue = Math.pow(2, count) - 1;
       customDecimalInput.max = maxValue;
       customDecimalInput.value = '';
       customDecimalInput.placeholder = `輸入 0-${maxValue} 的解鎖編號 (可選)`;
   };
   // 切換線路狀態
   // index: 相對於 currentWireCount 的索引 (0, 1, 2...)
   // globalIndex: 相對於 BIT_WEIGHTS 的索引 (0, 1, ..., 11)
   const toggleWireState = (index, originalColor, globalIndex) => {
       // 從 playerWireStates 獲取並更新狀態
       if (playerWireStates[index] === 1) {
           playerWireStates[index] = 0; // 變為 Cut (0)
       } else {
           playerWireStates[index] = 1; // 變為 Intact (1)
       }
       // 播放剪線/接回音效
       SoundManager.playWireToggle();

       const wireGroup = wireGrid.children[globalIndex];
       const wireElement = wireGroup.querySelector('.wire');
       const stateElement = wireGroup.querySelector('.player-state');

       if (playerWireStates[index] === 0) {
           // Visuals for Cut (0)
           wireElement.style.height = '40%';
           wireElement.style.backgroundColor = '#333';
           wireElement.style.borderTop = `3px dashed ${originalColor}`;
           stateElement.textContent = '0';
           stateElement.classList.remove('text-white');
           stateElement.classList.add('text-yellow-400');

       } else {
           // Visuals for Intact (1)
           wireElement.style.height = '100%';
           wireElement.style.backgroundColor = originalColor;
           wireElement.style.borderTop = 'none';
           stateElement.textContent = '1';
           stateElement.classList.remove('text-yellow-400');
           stateElement.classList.add('text-white');
       }
   };
   // 啟動倒數計時器
   const startTimer = (duration) => {
       let timeRemaining = duration;
       initialTime = duration;

       clearInterval(timerInterval);

       // 初次顯示
       timerValueDisplay.textContent = `${Math.floor(timeRemaining / 60).toString().padStart(1, '0')}:${(timeRemaining % 60).toString().padStart(2, '0')}`;
       timerInterval = setInterval(() => {
           timeRemaining--;
           timerValueDisplay.textContent = `${Math.floor(timeRemaining / 60).toString().padStart(1, '0')}:${(timeRemaining % 60).toString().padStart(2, '0')}`;
           if (timeRemaining <= 10 && timeRemaining > 0) { // 在剩餘 10 秒內觸發警報
               timerValueDisplay.classList.add('text-red-500');
               if (timeRemaining === 10) {
                   SoundManager.startTimerAlarm(); // 啟動緊急警報音效
               }
           } else {
               timerValueDisplay.classList.remove('text-red-500');
               if (timeRemaining > 10) {
                   SoundManager.stopTimerAlarm(); // 確保警報在 10 秒以上時停止
               }
           }
           if (timeRemaining <= 0) {
               clearInterval(timerInterval);
               gameOver(false, "警報：核心過載！計時歸零，任務失敗！");
           }
       }, 1000);
   };
   // 遊戲開始設置
   const initializeGame = (decimal) => {
       // 確保在開始前清除所有動畫視覺效果
       resetWireVisuals();

       const selectedWireCount = parseInt(wireCountSelect.value);
       currentWireCount = selectedWireCount;
       const maxValue = Math.pow(2, currentWireCount) - 1;
       initialTime = parseInt(timeSelect.value);

       // 1. 設定目標十進位數
       if (decimal !== null && decimal >= 0 && decimal <= maxValue) {
           targetDecimal = decimal;
       } else {
           targetDecimal = Math.floor(Math.random() * (maxValue + 1));
       }

       // TargetBinary 只基於 currentWireCount
       targetBinary = decimalToBinary(targetDecimal, currentWireCount);
       targetDecimalDisplay.textContent = targetDecimal;

       // 還原目標數字的字體樣式
       targetDecimalDisplay.classList.remove('text-xl', 'md:text-2xl', 'font-bold', 'text-gray-500');
       targetDecimalDisplay.classList.add('text-5xl', 'font-extrabold', 'text-red-500');
       // 2. 重置線路狀態 UI
       createWires(currentWireCount);

       // 3. 啟用遊戲並啟動計時器
       gameActive = true;
       defuseButton.disabled = false;
       defuseButton.classList.remove('opacity-50', 'cursor-not-allowed');
       startTimer(initialTime);
       SoundManager.stopTimerAlarm(); // 確保警報停止
       SoundManager.init(); // 確保 Audio Context 已啟動

       console.log(`[任務啟動] 解鎖編號 (Decimal): ${targetDecimal}, 正確指令序列 (Binary): ${targetBinary}`);
       // 隱藏設定面板，聚焦遊戲
       document.getElementById('setup-panel').classList.add('hidden');
       document.getElementById('game-container').classList.remove('opacity-50');
   };

   // 檢查並執行動畫
   const checkDefuse = () => {
       if (!gameActive) return;
       clearInterval(timerInterval);
       defuseButton.disabled = true;
       defuseButton.classList.add('opacity-50');

       // 開始動畫驗證，從最高位元開始 (相對索引 0)
       runValidationAnimation(0);
   };

   // 逐條線路驗證動畫
   const runValidationAnimation = (index) => {
       if (index >= currentWireCount) {
           // 全數檢查完畢，成功
           gameOver(true, "核心關閉成功！方舟城市得以延續！");
           return;
       }

       // 計算 globalIndex: 總線路數 - currentWireCount + index
       const globalIndex = BIT_WEIGHTS.length - currentWireCount + index;
       const wireGroup = wireGrid.children[globalIndex];

       const playerState = playerWireStates[index].toString();
       const targetBit = targetBinary[index];
       const isCorrect = playerState === targetBit;

       // 1. 設置閃爍效果並播放音效
       wireGroup.classList.add(isCorrect ? 'pulse-green' : 'pulse-red');
       isCorrect ? SoundManager.playCorrect() : SoundManager.playFail(); // 播放正確或錯誤音效

       // 2. 移除閃爍效果並決定下一步
       setTimeout(() => {
           // 清除動畫類別
           wireGroup.classList.remove('pulse-green', 'pulse-red');

           if (!isCorrect) {
               // 立即失敗並爆炸
               const weight = BIT_WEIGHTS[globalIndex];
               gameOver(false, `指令序列錯誤！第 ${index + 1} 條線 (權重: ${weight}) 設定錯誤！`);
           } else {
               // 答對，繼續檢查下一條線
               runValidationAnimation(index + 1);
           }
       }, 300); // 閃爍持續時間 300ms
   };

   // 重置所有線路組的視覺效果
   const resetWireVisuals = () => {
       Array.from(wireGrid.children).forEach(group => {
           group.classList.remove('pulse-green', 'pulse-red');
           group.style.boxShadow = 'none';
           group.style.transition = 'none';
       });
   };
   // 遊戲結束
   const gameOver = (success, message) => {
       gameActive = false;
       defuseButton.disabled = true;
       defuseButton.classList.add('opacity-50', 'cursor-not-allowed');
       timerValueDisplay.classList.remove('text-red-500');

       SoundManager.stopTimerAlarm(); // 停止所有警報

       if (success) {
           SoundManager.playSuccess(); // 播放成功音效
       } else {
           SoundManager.playExplosion(); // 播放爆炸音效
       }

       showCustomAlert(success, message, targetBinary);
       // 顯示設定面板以重新開始
       document.getElementById('setup-panel').classList.remove('hidden');
       document.getElementById('game-container').classList.add('opacity-50');
   };
   // 自定義訊息框 (取代 alert)
   const showCustomAlert = (success, message, correctBinary) => {
       messageBox.classList.remove('hidden');
       messageBox.classList.add('flex');

       // 根據結果設定樣式
       if (success) {
           messageContent.className = 'p-8 rounded-xl text-center shadow-2xl w-full max-w-sm bg-green-800 border-4 border-green-500 text-white';
       } else {
           messageContent.className = 'p-8 rounded-xl text-center shadow-2xl w-full max-w-sm bg-red-800 border-4 border-red-500 text-white';
       }
       // 構建內容
       let html = `<h2 class="text-4xl font-bold mb-4">${success ? '任務完成！' : '核心爆炸！'}</h2>`;
       html += `<p class="text-xl mb-4">${message}</p>`;

       const timerText = timerValueDisplay.textContent.split(':');
       // 計算總共經過的時間
       const secondsElapsed = initialTime - (parseInt(timerText[0]) * 60 + parseInt(timerText[1]));

       if (!success) {
           html += `<p class="mb-4 text-sm text-yellow-300">解鎖編號 (Decimal): ${targetDecimal}</p>`;
           html += `<p class="text-2xl font-mono mb-6 text-red-300">正確指令序列 (Binary): ${correctBinary.split('').join(' ')}</p>`;
       } else {
           html += `<p class="mb-4 text-sm text-yellow-300">解鎖編號 (Decimal): ${targetDecimal}</p>`;
           html += `<p class="text-2xl font-mono mb-6 text-green-300">反應時間: ${secondsElapsed} 秒</p>`;
       }

       html += `<button id="close-alert-button" class="mt-4 bg-gray-600 hover:bg-gray-700 py-2 px-6 rounded-lg text-white font-semibold">確定 / 重新部署</button>`;
       messageContent.innerHTML = html;

       document.getElementById('close-alert-button').onclick = () => {
           messageBox.classList.add('hidden');
           messageBox.classList.remove('flex');
       };
   };
   // --- 事件監聽器 ---

       defuseButton.addEventListener('click', checkDefuse);

       // 線路數選擇器變更時
   wireCountSelect.addEventListener('change', (e) => {
       createWires(e.target.value);
   });

       // 遊戲開始/設定按鈕
   startGameButton.addEventListener('click', () => {
       const maxValue = Math.pow(2, currentWireCount) - 1;
       let decimalValue = parseInt(customDecimalInput.value);

       // 檢查輸入是否在範圍內
       if (isNaN(decimalValue) || decimalValue < 0 || decimalValue > maxValue) {
           decimalValue = null; // 讓 initializeGame 隨機產生
       }
       initializeGame(decimalValue);
   });
   // 輸入框限制
   customDecimalInput.addEventListener('input', (e) => {
       const maxValue = Math.pow(2, currentWireCount) - 1;
       let val = parseInt(e.target.value);
       if (val > maxValue) e.target.value = maxValue;
       if (val < 0) e.target.value = 0;
   });
   // 初始化應用程式
   window.onload = () => {
       // 根據預設選擇 (12 條線) 創建線路
       createWires(wireCountSelect.value);

       // 初始狀態下，遊戲面板變暗，等待設定
       document.getElementById('game-container').classList.add('opacity-50');
       defuseButton.disabled = true;
       defuseButton.classList.add('opacity-50', 'cursor-not-allowed');
       // 初始化目標數字顯示為等待狀態
       targetDecimalDisplay.textContent = '等待任務啟動...';
       targetDecimalDisplay.classList.remove('text-5xl', 'font-extrabold', 'text-red-500');
       targetDecimalDisplay.classList.add('text-xl', 'md:text-2xl', 'font-bold', 'text-gray-500');
       console.log("遊戲已初始化。請在任務設置區選擇線路數、時間並設定解鎖編號以開始。");
   };
</script>
